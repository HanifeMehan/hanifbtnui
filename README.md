## Install

```bash
npm install --save hanifbtnui
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'hanifbtnui'
import 'hanifbtnui/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
